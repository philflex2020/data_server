#include <cassert>
#include <cstdio>
#include <chrono>
#include <filesystem>
#include <functional>
#include <string>
#include <vector>
namespace fs = std::filesystem;

// ── Tiny assertion/reporting helpers ──────────────────────────────────────

static int g_pass = 0, g_fail = 0;

#define EXPECT(cond, msg) do { \
    if (cond) { ++g_pass; printf("  PASS  %s\n", msg); } \
    else      { ++g_fail; printf("  FAIL  %s  (line %d)\n", msg, __LINE__); } \
} while(0)

#define EXPECT_EQ(a, b, msg) EXPECT((a) == (b), msg)

// ── Benchmark helper ──────────────────────────────────────────────────────

template<typename F>
double bench_mps(F fn, int iters = 1'000'000) {
    auto t0 = std::chrono::steady_clock::now();
    for (int i = 0; i < iters; ++i) fn();
    auto t1 = std::chrono::steady_clock::now();
    double ms = std::chrono::duration<double, std::milli>(t1-t0).count();
    return iters / (ms / 1000.0);
}

// ── Temp directory RAII ───────────────────────────────────────────────────

struct TempDir {
    fs::path path;
    TempDir() : path(fs::temp_directory_path() / "writer_test_XXXXXX") {
        char tmpl[256]; snprintf(tmpl, sizeof(tmpl), "%s", path.string().c_str());
        mkdtemp(tmpl); path = tmpl;
    }
    ~TempDir() { fs::remove_all(path); }
};

// ── Forward-declare writer.cpp internals ─────────────────────────────────
// (they are static — include via the PARQUET_WRITER_NO_MAIN compile unit)
// Access via the compiled object; declare extern for linkage where needed.

// ── Test functions ────────────────────────────────────────────────────────

void test_sanitize() {
    printf("[U1] sanitize_path_segment\n");
    // Direct calls — sanitize_path_segment is static; use parse_topic as proxy
    // or expose with a test-only declaration in writer.cpp:
    //   #ifdef PARQUET_WRITER_NO_MAIN
    //   std::string sanitize_path_segment(std::string);  // extern for tests
    //   #endif
    // Alternatively test indirectly via parse_topic:
    EXPECT(!parse_topic("/site=0").has_value(),   "empty source_type → nullopt");
    EXPECT(!parse_topic("../x").has_value(),       ".. source_type → nullopt");
    EXPECT( parse_topic("batteries").has_value(),  "normal → valid");
}

void test_parse_topic() {
    printf("[U2] parse_topic\n");
    {
        auto r = parse_topic("batteries/site=0/rack=1/module=2/cell=3/voltage");
        EXPECT(r.has_value(), "full topic parses");
        EXPECT_EQ(r->source_type, std::string("batteries"), "source_type");
        EXPECT_EQ(r->kv.at("site"), 0, "site=0");
        EXPECT_EQ(r->kv.at("cell"), 3, "cell=3");
        EXPECT_EQ(r->field_name, std::string("voltage"), "field_name=voltage");
    }
    EXPECT(!parse_topic("").has_value(), "empty topic → nullopt");
    EXPECT(!parse_topic("/").has_value(), "only slash → nullopt");
}

void test_validate_config() {
    printf("[U4] validate_config\n");
    Config good;
    EXPECT(validate_config(good), "default config is valid");

    Config bad1 = good; bad1.flush_interval_seconds = 0;
    EXPECT(!validate_config(bad1), "flush_interval=0 invalid");

    Config bad2 = good; bad2.compression = "bz2";
    EXPECT(!validate_config(bad2), "unknown compression invalid");

    Config bad3 = good; bad3.base_path = "";
    EXPECT(!validate_config(bad3), "empty base_path invalid");
}

void bench_parse_topic() {
    printf("[K1] parse_topic throughput\n");
    const std::string topic = "batteries/site=1/rack=2/module=3/cell=4/voltage";
    double mps = bench_mps([&]{ (void)parse_topic(topic); });
    printf("  %.2f M calls/sec  (%s)\n", mps / 1e6, mps > 2e6 ? "PASS" : mps > 500e3 ? "WARN" : "FAIL");
}

int main() {
    printf("=== writer.cpp unit tests ===\n\n");
    test_sanitize();
    test_parse_topic();
    test_validate_config();
    // ... more unit tests ...

    printf("\n=== KPI benchmarks ===\n\n");
    bench_parse_topic();
    // ... more benchmarks ...

    printf("\n=== Results: %d passed, %d failed ===\n", g_pass, g_fail);
    return g_fail == 0 ? 0 : 1;
}

Note on static functions: most writer functions are static (translation-unit scope). To test them directly, add a small block in writer.cpp guarded by #ifdef PARQUET_WRITER_NO_MAIN that re-declares them as extern, or move the function bodies to a shared writer_core.cpp and keep only main() in writer.cpp. The latter is the cleaner long-term approach (see minor improvement candidate).
